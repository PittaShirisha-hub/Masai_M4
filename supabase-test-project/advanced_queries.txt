========================================
Query 1: Count total number of students
========================================
SQL Query:
SELECT COUNT(*) AS total_students
FROM students;

Output:
total_students
--------------
10


========================================
Query 2: Average age of students
========================================
SQL Query:
SELECT AVG(age) AS average_age
FROM students;

Output:
average_age
-----------
22.5


========================================
Query 3: Minimum and Maximum age
========================================
SQL Query:
SELECT 
    MIN(age) AS minimum_age,
    MAX(age) AS maximum_age
FROM students;

Output:
minimum_age | maximum_age
------------|------------
18          | 26


========================================
Query 4: Count students enrolled in each course
========================================
SQL Query:
SELECT 
    course,
    COUNT(*) AS student_count
FROM students
GROUP BY course;

Output:
course             | student_count
-------------------|---------------
Web Development    | 4
Data Science       | 3
UI/UX Design       | 3


========================================
Query 5: Count students with age > 20
========================================
SQL Query:
SELECT COUNT(*) AS students_above_20
FROM students
WHERE age > 20;

Output:
students_above_20
-----------------
7


========================================
Query 6: Course-wise student count
========================================
SQL Query:
SELECT

Output:
course             | student_count
-------------------|---------------
Web Development    | 4
Data Science       | 3
UI/UX Design       | 3


========================================
Query 7: Courses with more than 1 student
========================================
SQL Query:
SELECT 
    course,
    COUNT(*) AS student_count
FROM students
GROUP BY course
HAVING COUNT(*) > 1;

Output:
course             | student_count
-------------------|---------------
Web Development    | 4
Data Science       | 3
UI/UX Design       | 3


========================================
Query 8: Courses sorted by highest student count
========================================
SQL Query:
SELECT 
    course,
    COUNT(*) AS student_count
FROM students
GROUP BY course
ORDER BY student_count DESC;

Output:
course             | student_count
-------------------|---------------
Web Development    | 4
Data Science       | 3
UI/UX Design       | 3
